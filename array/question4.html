Problem Statement

Given n non-negative integers representing the heights of vertical lines, find two lines that together with the x-axis form a container, such that the container contains the most water.

Example

Input: height = [1,8,6,2,5,4,8,3,7]
Output: 49
Explanation: The maximum container is formed between height[1] and height[8] with an area of 49.

Approach

To maximize the water contained, use the two-pointer technique. Start with two pointers, one at the beginning and one at the end. Calculate the area and move the pointer with the smaller height inwards. Repeat until the pointers meet.

This approach runs in O(n) time complexity.

Solution Code (C++)

#include <iostream>
#include <vector>
#include <algorithm> // For min and max functions

using namespace std;

// Function to calculate the maximum water container area
int maxArea(vector& height) {
    int left = 0, right = height.size() - 1;
    int max_water = 0;

    // Use two pointers: one at the start and one at the end
    while (left < right) {
        // Calculate area formed by current left and right pointers
        int minHeight = min(height[left], height[right]);
        int width = right - left;
        int area = minHeight * width;

        // Update max water container area
        max_water = max(max_water, area);

        // Move the pointer with the smaller height inward
        if (height[left] < height[right])
            left++;
        else
            right--;
    }
    
    return max_water;
}

int main() {
    // Example input
    vector height = {1, 8, 6, 2, 5, 4, 8, 3, 7};
    
    // Output the maximum water container area
    cout << "Maximum water: " << maxArea(height) << endl;
    
    return 0;
}
